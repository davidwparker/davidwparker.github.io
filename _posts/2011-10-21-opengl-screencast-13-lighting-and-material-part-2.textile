GL_COLOR_MATERIAL, GL_FRONT, GL_FRONT_AND_BACK, GL_AMBIENT, GL_DIFFUSE, GL_SPECULAR, GL_SHININESS, GL_EMISSION, GL_AMBIENT_AND_DIFFUSE
